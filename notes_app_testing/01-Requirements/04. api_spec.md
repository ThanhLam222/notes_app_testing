### Note App's API Specification
```yaml

info: 
  title: The Note App APIs
  description: The Note App APIs
  version: "1.0.0"

servers:
  - url: http://localhost:4000
tags:
  - name: Index
    description: APIs for the landing and about pages.

  - name: Authentication
    description: APIs for user management.

  - name: Note
    description: APIs for managing user notes.

components:
  securitySchemes:
    cookieAuth:
      type: apiKey
      in: cookie
      name: connect.sid
  schemas:
    NewNoteAndEditNote:
      type: object
      properties:
        title:
          type: string
        description:
          type: string
      required:
        - title
        - description
    UserAuth:
      type: object
      properties:
        email:
          type: string
          format: email
        password:
          type: string
          minLength: 4
      required:
        - email
        - password
    SignUpForm:
      type: object
      properties:
        name:
          type: string
        email:
          type: string
          format: email
        password:
          type: string
          minLength: 4
        confirm_password:
          type: string
      required:
        - email
        - password
        - confirm_password

paths:
  /:
    get:
      tags:
        - Index
      summary: Index page
      description: The starting point of unauthenticated users.
      responses:
        200:
          description: Successful response - Renders the Index page
          headers:
            Set-Cookie:
              description: "Sent every request. But session ID only change in case of sign in or log out"
              schema:
                type: string
                example: connect.sid=s%3A...
          content:
            text/html:
              example: "<!DOCTYPE html>...</html>"
    
/about:
    get:
      tags:
        - Index
      summary: About page
      description: Renders the "About" page with information about the app.
      responses:
        200:
          description: Successful response - Renders the About page
          headers:
            Set-Cookie:
              description: "Sent every request. But session ID only change in case of sign in or log out"
              schema:
                type: string
                example: connect.sid=s%3A...
          content:
            text/html:
              example: "<!DOCTYPE html>...</html>"

/auth/signup:
    get:
      tags:
        - Authentication
      summary: Sign up page
      description: Renders the Sign up form
      responses:
        200:
          description: Renders the Sign up page - Contains flash messages if redirected from signup.
          headers:
            Set-Cookie:
              description: "Sent every request. But session ID only change in case of sign in or log out"
              schema:
                type: string
                example: connect.sid=s%3A...
          content:
            text/html:
              example: "<!DOCTYPE html>...</html>"

    post:
      tags:
        - Authentication
      summary: Submit sign up form
      description: Register a new user. The email must be unique.
      requestBody:
        content:
          application/x-www-form-urlencoded: 
            schema:
              $ref: "#/components/schemas/SignUpForm"
      responses:
        302:
          description: Redirects to Sign in page on success or to Sign up page on failure of checking email existing with flash message
          headers:
            Set-Cookie:
              description: "Sent every request. But session ID only change in case of sign in or log out"
              schema:
                type: string
                example: connect.sid=s%3A...
            Location:
              schema:
                type: string
                enum:
                  - /auth/signin
                  - /auth/signup

        200:
          description: Renders the Sign up page with previously filled data (with an array of error messages) if basic data validation fails.
          headers:
            Set-Cookie:
              description: "Sent every request. But session ID only change in case of sign in or log out"
              schema:
                type: string
                example: connect.sid=s%3A...
          content:
            text/html:
              example: "<!DOCTYPE html>...</html>"        

/auth/signin:
    get:
      tags:
        - Authentication
      summary: Sign in page
      description: Renders the Sign in form
      responses:
        200:
          description: Renders the Sign in page - Contains flash messages if redirected from signup or logout.
          headers:
            Set-Cookie:
              description: "Sent every request. But session ID only change in case of sign in or log out"
              schema:
                type: string
                example: connect.sid=s%3A...
          content:
            text/html:
              example: "<!DOCTYPE html>...</html>"

    post:
      tags:
        - Authentication
      summary: Submit sign in form
      description: Authenticates a user
      requestBody:
      content:
        application/x-www-form-urlencoded: 
          schema:
            $ref: "#/components/schemas/UserAuth"
      responses:
        302:
          description: Redirects to /notes on success or to /auth/signin on failure (with error flash message)
          headers:
            Set-Cookie:
              description: "Sent every request. But session ID only change in case of sign in or log out"
              schema:
                type: string
                example: connect.sid=s%3A...
            Location:
              schema:
                type: string
                enum:
                  - /notes
                  - /auth/signin

/auth/logout:
    get:
      tags:
        - Authentication
      summary: Logs out the user
      description: User log out of the **Note App**
      security:
        - cookieAuth: []
      responses:
        302:
          description: Always redirects to the Sign in page, with a success flash message.
          headers:
            Set-Cookie:
              description: "Sent every request. But session ID only change in case of sign in or log out"
              schema:
                type: string
                example: connect.sid=s%3A...
            Location:
              schema:
                type: string
                example: /auth/signin

/notes:
    get:
      tags:
        - Note
      summary: All notes page
      description: Displays a list of all notes for the authenticated user.
      security:
        - cookieAuth: []
      responses:
        200:
          description: Renders the notes page with a list of notes. Flash messages from previous actions (e.g., add/edit/delete) are displayed here.
          headers:
            Set-Cookie:
              description: "Sent every request. But session ID only change in case of sign in or log out"
              schema:
                type: string
                example: connect.sid=s%3A...
          content:
            text/html:
              example: "<!DOCTYPE html>...</html>"

        302:
          description: Redirects to Sign in page if not authenticated, with a flash message.
          headers:
            Set-Cookie:
              description: "Sent every request. But session ID only change in case of sign in or log out"
              schema:
                type: string
                example: connect.sid=s%3A...
            Location:
              schema:
                type: string
                example: /auth/signin

/notes/add:
    get:
      tags:
        - Note
      summary: New note page
      description: Renders the form to create new note
      security:
        - cookieAuth: []      
      responses:
        200:
          description: Renders the New note page
          headers:
            Set-Cookie:
              description: "Sent every request. But session ID only change in case of sign in or log out"
              schema:
                type: string
                example: connect.sid=s%3A...
          content:
            text/html:
              example: "<!DOCTYPE html>...</html>"

        302:
          description: Redirects to Sign in page if not authenticated, with a flash message.
          headers:
            Set-Cookie:
              description: "Sent every request. But session ID only change in case of sign in or log out"
              schema:
                type: string
                example: connect.sid=s%3A...
            Location:
              schema:
                type: string
                example: /auth/signin

/notes/new-note:
    post:
      tags:
        - Note
      summary: Submit new note form
      description: Creates a new note to save user's thought.
      security:
        - cookieAuth: []
      requestBody:
        content:
          application/x-www-form-urlencoded: 
            schema:
              $ref: "#/components/schemas/NewNoteAndEditNote"
      responses:
        302:
          description: Redirects to /notes on success or to /auth/signin on authentication error with flash message
          headers:
            Set-Cookie:
              description: "Sent every request. But session ID only change in case of sign in or log out"
              schema:
                type: string
                example: connect.sid=s%3A...
            Location:
              schema:
                type: string
                enum:
                  - /notes
                  - /auth/signin
        200:
          description: Renders the New note page with previously filled data and an array of error messages on data validation error
          headers:
            Set-Cookie:
              description: "Sent every request. But session ID only change in case of sign in or log out"
              schema:
                type: string
                example: connect.sid=s%3A...
          content:
            text/html:
              example: "<!DOCTYPE html>...</html>"



/notes/edit/:id:
    get:
      tags:
        - Note
      summary: Edit note page
      description: Renders the form to edit an existing note
      security:
        - cookieAuth: []
      parameters:
        - name: id
          in: path
          required: true
          schema:
            type: string
      responses:
        200:
          description: Renders the 'Edit Note' form with existing data.
          headers:
            Set-Cookie:
              description: "Sent every request. But session ID only change in case of sign in or log out"
              schema:
                type: string
                example: connect.sid=s%3A...
          content:
            text/html:
              example: "<!DOCTYPE html>...</html>"    
        302:
          description: Redirects to /notes on authorization or to /auth/signin on authentication errors, with flash messages explaining the failure.
          headers:
            Set-Cookie:
              description: "Sent every request. But session ID only change in case of sign in or log out"
              schema:
                type: string
                example: connect.sid=s%3A...
            Location:
              schema:
                type: string
                enum:
                  - /notes
                  - /auth/signin

/notes/edit-note/:id:
    put:
      tags:
        - Note
      summary: Submit edit note form
      description: Updates an existing note of user
      security:
        - cookieAuth: []
      parameters:
        - name: id
          in: path
          required: true
          schema:
            type: string
      requestBody:
        content:
          application/x-www-form-urlencoded: 
            schema:
              $ref: "#/components/schemas/NewNoteAndEditNote"
      responses:
        302:
          description: Redirects to /notes on success and on authorization error or to /auth/signin on authentication error. The flash message indicates the outcome.
          headers:
            Set-Cookie:
              description: "Sent every request. But session ID only change in case of sign in or log out"
              schema:
                type: string
                example: connect.sid=s%3A...
            Location:
              schema:
                type: string
                enum:
                  - /notes
                  - /auth/signin

        200:
          description: Renders the Edit note page again on data validation failure, with filled data and an array of error messages.
          headers:
            Set-Cookie:
              description: "Sent every request. But session ID only change in case of sign in or log out"
              schema:
                type: string
                example: connect.sid=s%3A...
          content:
            text/html:
              example: "<!DOCTYPE html>...</html>"

/notes/delete/:id:
    delete:
      tags:
        - Note
      summary: Delete note
      description: Delete a specific note of user
      security:
        - cookieAuth: []
      parameters:
        - name: id
          in: path
          required: true
          schema:
            type: string
      responses:
        302:
          description: Redirects to /notes on success and on authorization error or to /auth/signin on authentication error. The flash message indicates the outcome.
          headers:
            Set-Cookie:
              description: "Sent every request. But session ID only change in case of sign in or log out"
              schema:
                type: string
                example: connect.sid=s%3A...
            Location:
              schema:
                type: string
                enum:
                  - /notes
                  - /auth/signin
```
